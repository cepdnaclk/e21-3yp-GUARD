#include <DallasTemperature.h>
#include <WiFiUdp.h>
#include "time.h"

// ---------------- WiFi Reset -----------------------
#define RESET_BUTTON 0
#define RESET_DURATION 3000
unsigned long pressStart = 0;
//---------------------------------------------------

// ---------------- Sensor & Timing -----------------
unsigned long lastTempTime = 0;
unsigned long lastReconnectAttempt = 0;
const unsigned long TEMP_INTERVAL = 10000; // 10 seconds
//---------------------------------------------------

// ---------------- Device Data ---------------------
unsigned long DeviceID = 100;
String clientID = "ESP32_" + String(DeviceID);

// MQTT Credentials (Update if necessary)
const char* mqtt_server = "192.168.1.5";
const char* mqtt_user = "Admin";         
const char* mqtt_password = "B88vf9kp:}Xj"; 
//---------------------------------------------------

// ---------------- DS18B20 Setup -------------------
#define ONE_WIRE_BUS 4 // GPIO4 connected to DQ
OneWire oneWire(ONE_WIRE_BUS);
DallasTemperature sensors(&oneWire);
//---------------------------------------------------

// ---> THE FIX: Function Prototype for PlatformIO <---
void publishSensor(String parameter, float value);

// ---------------- MQTT Setup ----------------------
WiFiClient espClient;
PubSubClient client(espClient);

String buildTopic(String parameter) {
  return "sensor/" + String(DeviceID) + "/" + parameter;
}

String commandTopic() {
  return "device/" + String(DeviceID) + "/command";
}

// Reconnect function returns a boolean so the main loop knows if it succeeded
boolean reconnect() {
  Serial.print("[INFO] Attempting MQTT connection to ");
  Serial.print(mqtt_server);
  Serial.print("... ");

  if (client.connect(clientID.c_str(), mqtt_user, mqtt_password)) {
    Serial.println("CONNECTED!");
    
    // Subscribe to commands once connected
    if(client.subscribe(commandTopic().c_str())) {
       Serial.println("[INFO] Subscribed to command topic: " + commandTopic());
    } else {
       Serial.println("[ERROR] Failed to subscribe to command topic!");
    }
    return true;
  } else {
    Serial.print("FAILED. State/Error Code = ");
    Serial.println(client.state());
    Serial.println("[WARN] Will try again in 5 seconds.");
    return false;
  }
}

// ---------------- MQTT Callback ------------------
void callback(char* topic, byte* payload, unsigned int length) {
  String message;
  for (unsigned int i = 0; i < length; i++) {
    message += (char)payload[i];
  }

  Serial.println("[INFO] Message arrived on topic: " + String(topic));
  Serial.println("[INFO] Payload: " + message);

  if (message == "temperature") {
    Serial.println("[INFO] Manual temperature request received.");
    sensors.requestTemperatures();
    float tempC = sensors.getTempCByIndex(0);
    
    if(tempC == DEVICE_DISCONNECTED_C) {
      Serial.println("[ERROR] Sensor disconnected! Cannot publish manual request.");
    } else {
      publishSensor("temperature", tempC);
    }
  } else {
    Serial.println("[WARN] Unknown command received.");
  }
}
//---------------------------------------------------

// ---------------- NTP Time ------------------------
const char* ntpServer = "pool.ntp.org";
const long gmtOffset_sec = 19800; // Sri Lanka GMT+5:30
const int daylightOffset_sec = 0;

void initTime() {
  Serial.println("[INFO] Initializing NTP Time...");
  configTime(gmtOffset_sec, daylightOffset_sec, ntpServer);
  
  // Wait up to 10 seconds to get the time
  struct tm timeinfo;
  int retryCounter = 0;
  while (!getLocalTime(&timeinfo) && retryCounter < 10) {
    Serial.print(".");
    delay(1000);
    retryCounter++;
  }
  
  if (retryCounter >= 10) {
    Serial.println("\n[ERROR] Failed to obtain time from NTP server!");
  } else {
    Serial.println("\n[INFO] Time successfully synchronized.");
  }
}

String getTimeStamp() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    return "00-00-0000 00:00:00"; // Fallback if time fails
  }
  char buf[25];
  strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", &timeinfo);
  return String(buf);
}
//---------------------------------------------------

// ---------------- Publish Sensor ------------------
void publishSensor(String parameter, float value) {
  String topic = buildTopic(parameter);

  // Construct JSON carefully
  String payload = "{\"value\":";
  payload += String(value);
  payload += ",\"time\":\"";
  payload += getTimeStamp();
  payload += "\"}";

  Serial.print("[INFO] Publishing to ");
  Serial.print(topic);
  Serial.print(" -> ");
  Serial.println(payload);

  // Check if publish was successful, using retain flag 'true'
  if (client.publish(topic.c_str(), payload.c_str(), true)) {
    Serial.println("[INFO] Publish Success.");
  } else {
    Serial.println("[ERROR] Publish Failed! Payload may be too large or connection lost.");
  }
}
//---------------------------------------------------

void setup() {
  Serial.begin(9600);
  delay(1000);
  Serial.println("\n\n===================================");
  Serial.println("[INFO] Device Starting Up");
  Serial.println("===================================");

  pinMode(RESET_BUTTON, INPUT_PULLUP);

  // ---------------- WiFi Setup -------------------
  Serial.println("[INFO] Connecting to WiFi via WiFiManager...");
  WiFiManager wm;
  if (wm.autoConnect("ESP32_Setup_100")) {
    Serial.println("[INFO] WiFi Connected!");
    Serial.print("[INFO] IP Address: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("[ERROR] WiFiManager failed to connect and hit timeout.");
    // Wait a bit and restart
    delay(3000);
    ESP.restart();
  }

  // ---------------- NTP Setup --------------------
  initTime();

  // ---------------- DS18B20 ----------------------
  Serial.println("[INFO] Initializing Temperature Sensor...");
  sensors.begin();
  int deviceCount = sensors.getDeviceCount();
  if (deviceCount == 0) {
    Serial.println("[ERROR] NO DS18B20 SENSORS FOUND! Check wiring to GPIO4.");
  } else {
    Serial.print("[INFO] Found ");
    Serial.print(deviceCount);
    Serial.println(" temperature sensor(s).");
  }

  // ---------------- MQTT Setup -------------------
  client.setServer(mqtt_server, 1883);
  client.setCallback(callback);
  // Initial connection attempt
  lastReconnectAttempt = 0;
}

//---------------------------------------------------
void loop() {
  // ---------------- Non-Blocking MQTT Connection ---
  if (!client.connected()) {
    unsigned long now = millis();
    // Try to reconnect only once every 5 seconds to prevent freezing
    if (now - lastReconnectAttempt > 5000) {
      lastReconnectAttempt = now;
      if (reconnect()) {
        lastReconnectAttempt = 0;
      }
    }
  } else {
    // Client is connected, process incoming messages
    client.loop();
  }

  // ---------------- Sensor Auto-Publish -----------
  unsigned long now = millis();
  if (now - lastTempTime > TEMP_INTERVAL) {
    lastTempTime = now;
    
    // Only attempt to read and publish if connected to MQTT
    if (client.connected()) {
      sensors.requestTemperatures();
      float tempC = sensors.getTempCByIndex(0);
      
      // Protect against sending completely bogus data if the sensor unplugs
      if(tempC == DEVICE_DISCONNECTED_C) {
         Serial.println("[ERROR] Failed to read temperature. Sensor disconnected!");
      } else {
         publishSensor("temperature", tempC);
      }
    }
  }

  // ---------------- WiFi Reset Button -------------
  if (digitalRead(RESET_BUTTON) == LOW) {
    if (pressStart == 0) pressStart = millis();
    if (millis() - pressStart >= RESET_DURATION) {
      Serial.println("[WARN] Reset Button held. Wiping WiFi settings and restarting...");
      WiFiManager wm;
      wm.resetSettings();
      delay(1000);
      ESP.restart();
    }
  } else {
    pressStart = 0;
  }
}
